import h2o
from h2o.frame import H2OFrame
import random

# Iniciamos el cluster de H2O
h2o.init()

# Creamos simulación de lanzamientos de dado (ejemplo: 1000 lanzamientos)
n_lanzamientos = 1000
dados = [[random.randint(1, 6)] for _ in range(n_lanzamientos)]  # Lista de listas

# Pasamos la lista a un H2OFrame
hf = H2OFrame(dados, column_names=["lanzamiento"])

# Calcular cuántas veces salió el número 5
conteo_5 = hf[hf["lanzamiento"] == 5].nrow

# Calcular probabilidad
probabilidad_5 = conteo_5 / n_lanzamientos

print(f"Número de lanzamientos: {n_lanzamientos}")
print(f"Veces que salió 5: {conteo_5}")
print(f"Probabilidad estimada de 5: {probabilidad_5:.4f}")

# Estadísticas básicas con H2O
print("\nResumen estadístico del experimento:")
print(hf.describe())
